001package org.xbib.elasticsearch.rest.csv;
002
003import org.elasticsearch.action.search.SearchResponse;
004import org.elasticsearch.common.bytes.BytesArray;
005import org.elasticsearch.common.collect.Sets;
006import org.elasticsearch.common.settings.loader.JsonSettingsLoader;
007import org.elasticsearch.rest.BytesRestResponse;
008import org.elasticsearch.rest.RestChannel;
009import org.elasticsearch.rest.RestResponse;
010import org.elasticsearch.rest.action.support.RestResponseListener;
011import org.elasticsearch.search.SearchHit;
012import org.xbib.elasticsearch.common.csv.CSVGenerator;
013
014import java.io.StringWriter;
015import java.util.List;
016import java.util.Map;
017import java.util.Set;
018
019import static org.elasticsearch.common.collect.Lists.newLinkedList;
020
021public class CSVToXContentListener extends RestResponseListener<SearchResponse> {
022
023    private Set<String> keys;
024
025    public CSVToXContentListener(RestChannel channel, Set<String> keys) {
026        super(channel);
027        this.keys = keys;
028    }
029
030    @Override
031    public final RestResponse buildResponse(SearchResponse response) throws Exception {
032        StringWriter writer = new StringWriter();
033        if (response.getHits() != null && response.getHits().getHits() != null && response.getHits().getHits().length > 0) {
034            CSVGenerator gen = new CSVGenerator(writer);
035            boolean fieldNamesWritten = false;
036            JsonSettingsLoader settingsLoader = new JsonSettingsLoader();
037            for (SearchHit hit : response.getHits().getHits()) {
038                Map<String,String> map = settingsLoader.load(hit.sourceRef().toUtf8());
039                if (!fieldNamesWritten) {
040                    List<String> list = newLinkedList();
041                    if (keys == null) {
042                        list.addAll(map.keySet());
043                    } else {
044                        list.addAll(keys);
045                    }
046                    gen.keys(list);
047                    gen.writeKeys();
048                    fieldNamesWritten = true;
049                }
050                if (keys == null) {
051                    for (String key : map.keySet()) {
052                        gen.write(map.get(key));
053                    }
054                } else {
055                    Set<String> nonexisting = Sets.difference(keys, map.keySet());
056                    for (String key : keys) {
057                        if (map.keySet().contains(key)) {
058                            gen.write(map.get(key));
059                        } else if (nonexisting.contains(key)) {
060                            gen.write(null);
061                        }
062                    }
063                }
064                gen.nextRow();
065            }
066            gen.close();
067        }
068        return new BytesRestResponse(response.status(), "text/csv", new BytesArray(writer.toString()), true);
069    }
070
071}
