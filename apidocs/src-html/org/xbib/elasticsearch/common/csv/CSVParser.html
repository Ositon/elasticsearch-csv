001package org.xbib.elasticsearch.common.csv;
002
003import java.io.IOException;
004import java.io.Reader;
005import java.util.Iterator;
006import java.util.LinkedList;
007import java.util.List;
008import java.util.NoSuchElementException;
009
010public class CSVParser {
011
012    private final CSVLexer lexer;
013
014    private final List<String> row;
015
016    private final Token reusableToken;
017
018    public CSVParser(Reader reader) throws IOException {
019        lexer = new CSVLexer(new LookAheadReader(reader), ',', '"', '"', '#', true, true);
020        row = new LinkedList<String>();
021        reusableToken = new Token();
022    }
023
024    public void close() throws IOException {
025        lexer.close();
026    }
027
028    public long getCurrentLineNumber() {
029        return lexer.getCurrentLineNumber();
030    }
031
032    public Iterator<List<String>> iterator() {
033        return new Iterator<List<String>>() {
034            private List<String> current;
035
036            private List<String> getNextRow() {
037                try {
038                    return CSVParser.this.nextRow();
039                } catch (final IOException e) {
040                    throw new RuntimeException(e);
041                }
042            }
043
044            public boolean hasNext() {
045                if (current == null) {
046                    current = getNextRow();
047                }
048                return current != null;
049            }
050
051            public List<String> next() {
052                List<String> next = current;
053                current = null;
054                if (next == null) {
055                    next = getNextRow();
056                    if (next == null) {
057                        throw new NoSuchElementException("no more rows");
058                    }
059                }
060                return next;
061            }
062
063            public void remove() {
064                throw new UnsupportedOperationException();
065            }
066        };
067    }
068
069    protected List<String> nextRow() throws IOException {
070        row.clear();
071        StringBuilder sb = null;
072        do {
073            reusableToken.reset();
074            lexer.nextToken(reusableToken);
075            String s = reusableToken.content.toString();
076            switch (reusableToken.type) {
077                case TOKEN:
078                    row.add(s);
079                    break;
080                case EORECORD:
081                    row.add(s);
082                    break;
083                case EOF:
084                    if (reusableToken.isReady) {
085                        row.add(s);
086                    }
087                    break;
088                case INVALID:
089                    throw new IOException("(line " + getCurrentLineNumber() + ") invalid parse sequence");
090                case COMMENT:
091                    if (sb == null) {
092                        sb = new StringBuilder();
093                    } else {
094                        sb.append(Constants.LF);
095                    }
096                    sb.append(reusableToken.content);
097                    reusableToken.type = Token.Type.TOKEN;
098                    break;
099                default:
100                    throw new IllegalStateException("unexpected token type: " + reusableToken.type);
101            }
102        } while (reusableToken.type == Token.Type.TOKEN);
103        return row;
104    }
105
106}
